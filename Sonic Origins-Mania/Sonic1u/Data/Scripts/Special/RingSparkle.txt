event ObjectUpdate
	if object.timer == 0
		if object.frame1 == 0
			object.frame2 = 15
			object.frame3 = 21
			
			temp2 = 16 // Used for pos offset randomness
			if object.propertyValue == 1
				temp2 = 24
			end if
			if object.propertyValue == 2
				temp2 = 24
			end if
			temp4 = temp2
			temp4 /= 2
			
			Rand(temp0, temp2)
			temp0 -= temp4
			Rand(temp1, temp2)
			temp1 -= temp4
			temp0 <<= 16
			temp1 <<= 16
			object.sparkle1X = temp0
			object.sparkle1Y = temp1
			
			if object.propertyValue != 1
				Rand(temp0, temp2)
				temp0 -= temp4
				Rand(temp1, temp2)
				temp1 -= temp4
				temp0 <<= 16
				temp1 <<= 16
				object.sparkle2X = temp0
				object.sparkle2Y = temp1
							
				Rand(temp0, temp2)
				temp0 -= temp4
				Rand(temp1, temp2)
				temp1 -= temp4
				temp0 <<= 16
				temp1 <<= 16
				object.sparkle3X = temp0
				object.sparkle3Y = temp1
			end if
		end if
	end if

	if object.timer == 1
		object.timer = 0
		object.frame1++
		
		if object.frame2 == 20
			object.frame2 = 29
		end if
		if object.frame2 != 29
			object.frame2++
		end if
		
		if object.frame3 != 29
			object.frame3++
		end if

		if object.frame1 == 15
			object.type = TypeName[Blank Object]
			object.frame = 0
		end if
	end if
	object.timer++
	
	if object.propertyValue == 1
		object.sparkle1Y += object.yvel
	end if
end event


event ObjectDraw
	CallFunction(SpecialSetup_GetBlockPos)
	temp0 += object.sparkle1X
	temp1 += object.sparkle1Y
	DrawSpriteXY(object.frame1, temp0, temp1)
	CallFunction(SpecialSetup_GetBlockPos)
	temp0 += object.sparkle2X
	temp1 += object.sparkle2Y
	DrawSpriteXY(object.frame2, temp0, temp1)
	CallFunction(SpecialSetup_GetBlockPos)
	temp0 += object.sparkle3X
	temp1 += object.sparkle3Y
	DrawSpriteXY(object.frame3, temp0, temp1)
end event
