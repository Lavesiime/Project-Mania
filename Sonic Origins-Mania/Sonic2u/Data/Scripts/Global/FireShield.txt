		break

		object.animationTimer = 0
		object.timer = 0
		object.state++
		// [Fallthrough]
	case FIRESHIELD_DASH
		object.animationTimer++
		object.frame++
		object.sortedDrawOrder = 0
		if object.frame < 38

		end if
		if object.frame > 49

		end if
		object.timer++
		if object.timer >= 24
			object.state = FIRESHIELD_IDLE_SETUP
		end if
		break

	end switch
