private alias object.value0 : object.timer

private alias 0 : BBSHOT_CHARGE1 // frame 1
private alias 1 : BBSHOT_CHARGE2 // frame 2
private alias 2 : BBSHOT_SHOT	 // actually shot the projectile

event ObjectUpdate
	switch object.state
	case BBSHOT_CHARGE1
		if object.timer < 8
			object.timer++
			object.animationTimer++
			if object.animationTimer >= 2
				object.frame++
				object.animationTimer = 0
			end if
		else
			object.timer = 0
			object.state++
		end if
		break

	case BBSHOT_CHARGE2
		if object.timer < 8
			if object.frame < 4
				object.frame = 4
			end if
			object.timer++
			object.animationTimer++
			if object.animationTimer >= 2
				object.frame++
				object.animationTimer = 0
			end if
			if object.frame > 5
				object.frame = 4
			end if
		else
			object.timer = 0
			object.state++
		end if
		break

	case BBSHOT_SHOT
		object.xpos += object.xvel
		object.ypos += object.yvel
		if object.frame < 6
			object.frame = 6
		end if
		object.timer++
		if object.timer >= 2
			object.frame++
			object.timer = 0
		end if
		if object.frame > 11
			object.frame = 6
		end if
		break
		
	end switch

	if object.outOfBounds == true
		object.type = TypeName[Blank Object]
	end if

	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		if object.state > BBSHOT_CHARGE1
			BoxCollisionTest(C_TOUCH, object.entityPos, -6, -6, 6, 6, currentPlayer, C_BOX, C_BOX, C_BOX, C_BOX)
			if checkResult == true
				CallFunction(Player_ProjectileHit)
			end if
		end if
	next
end event


event ObjectDraw
	object.inkEffect = INK_ADD
	object.alpha = 180
	temp0 = object.frame
	if object.direction == 1
		temp0 += 12
	end if
	DrawSpriteFX(temp0, FX_INK, object.xpos, object.ypos)
end event
