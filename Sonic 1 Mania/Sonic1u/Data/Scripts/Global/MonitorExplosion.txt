private alias object.value0 : object.timer


event ObjectUpdate
	object.timer++
	if object.timer > 2
		object.timer = 0
		
		object.frame++
		if object.frame > 12
			object.type = TypeName[Blank Object]
			object.frame = 0
		end if
	end if
end event


event ObjectDraw
	DrawSprite(object.frame)
end event
