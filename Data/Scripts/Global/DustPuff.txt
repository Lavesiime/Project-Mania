private alias object.propertyValue 		: object.targetPlayer

private alias object.xpos 				: player.xpos

private alias object.animation 			: player.animation
private alias object.collisionBottom 	: player.collisionBottom


// ========================
// Events
// ========================

event ObjectUpdate
	switch object.state
	case 0
		object.frame++
		if object.frame > 13
			object.frame = 0
		end if
		break
		
	case 1
		if object.frame < 14
			object.frame = 14
		end if
	
		object.animationTimer++
		if object.animationTimer > 2
			object.animationTimer = 0
			object.frame++
		end if
		if object.frame > 23
			object.type = TypeName[Blank Object]
		end if
		break
	
	case 2
		if object.frame < 24
			object.frame = 24
		end if
	
		object.animationTimer++
		if object.animationTimer > 1
			object.animationTimer = 0
			object.frame++
		end if
		if object.frame > 33
			object.type = TypeName[Blank Object]
		end if
		break
		
	end switch
end event


event ObjectDraw
	switch object.state
	case 0
		currentPlayer = object.targetPlayer
		object.direction = player[currentPlayer].direction
		object.xpos = player[currentPlayer].xpos
		object.iypos = player[currentPlayer].collisionBottom
		object.ypos += player[currentPlayer].ypos
		if player[currentPlayer].animation != ANI_SPINDASH
			object.type = TypeName[Blank Object]
		else
			DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
		end if
		break
		
	case 1
	case 2
		DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
		break
	
	end switch
end event
