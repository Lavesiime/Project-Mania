event ObjectUpdate
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -2, -16, 2, 16, currentPlayer, C_BOX, C_BOX, C_BOX, C_BOX)
	
		if checkResult == true

			object[currentPlayer].state = Player_State_Air
			object[currentPlayer].gravity = GRAVITY_AIR
			object[currentPlayer].animation = ANI_WATERSLIDE
			object[currentPlayer].yvel = -0x0B0000
			object[currentPlayer].speed = 0
			object[currentPlayer].xvel = 0
			object[currentPlayer].value1 = 0
			if object.state == 0
				object.state = 1
				CreateTempObject(TypeName[Water Geyser], 0, object.xpos, object.ypos)
				PlaySfx(SfxName[Water Geyser], 0)
			end if
			if object.timer < 20
				object.ypos -= 0x0B0000
				object.ypos -= object.yvel
			end if
		else
			if object.timer > 0
				object.ypos += 0x020000
				object.ypos += object.yvel
				if object.timer > 35
					object.type = TypeName[Blank Object]
				end if
			end if
		end if
	next
	if object.state == 1
		object.timer++
		if object.timer == 64
			object.timer = 0
		end if
		StopSfx(SfxName[Water Splash])
	end if
end event
